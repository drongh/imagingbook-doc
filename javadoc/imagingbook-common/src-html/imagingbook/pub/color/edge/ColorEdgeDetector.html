010package imagingbook.pub.color.edge;
011
012import ij.process.ByteProcessor;
013import ij.process.ColorProcessor;
014import ij.process.FloatProcessor;
015
016/**
017 * This is the common super-class for all color edge detectors.
018 * @author W. Burger
019 * @version 2013/05/30
020 */
021public abstract class ColorEdgeDetector {
022        
023        public abstract FloatProcessor getEdgeMagnitude();
024        public abstract FloatProcessor getEdgeOrientation();
025        
026        FloatProcessor getRgbFloatChannel(ColorProcessor cp, int c) {   // c = 0,1,2
027                int w = cp.getWidth();
028                int h = cp.getHeight();
029                ByteProcessor bp = new ByteProcessor(w, h, cp.getChannel(c + 1)); // numbered from 1,...,3!
030                return bp.convertToFloatProcessor();
031        }
032}
