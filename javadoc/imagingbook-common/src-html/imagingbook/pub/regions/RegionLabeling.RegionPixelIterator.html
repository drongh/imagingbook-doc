033/**
034 * This class does the complete region labeling for a given image.
035 * It is abstract, because the implementation of some parts depend
036 * upon the region labeling algorithm being used.
037 * Updated/checked: 2014-11-12
038 */
039public abstract class RegionLabeling {
040        
041        static final int BACKGROUND = 0;
042        static final int FOREGROUND = 1;
043        static final int START_LABEL = 2;
044
045        protected ImageProcessor ip;
046        protected int width;
047        protected int height;
048        protected int currentLabel;
049        protected int maxLabel; // the maximum label in the labels array
050        
051        protected int[][] labelArray;
052        // label values in labelArray can be:
053        //  0 ... unlabeled
054        // -1 ... previously visited background pixel
055        // >0 ... valid label
056        
057        protected List<BinaryRegion> regions;
058        
059        RegionLabeling(ByteProcessor ip) {
060                this.ip = ip;
061                width  = ip.getWidth();
062                height = ip.getHeight();
063                initialize();
064                applyLabeling();
065                collectRegions();
066        }
067        
068        void initialize() {
069                // set all pixels to either FOREGROUND or BACKGROUND (by thresholding)
070                labelArray = new int[width][height];
071                for (int v = 0; v < height; v++) {
072                        for (int u = 0; u < width; u++) {
073                                labelArray[u][v] = (ip.getPixel(u, v) > 0) ? FOREGROUND : BACKGROUND;
074                        }
075                }
076        }
077        
078        public List<BinaryRegion> getRegions() {
079                return getRegions(false);       // unsorted
080        }
081        
082        public List<BinaryRegion> getRegions(boolean sort) {
083                if (regions == null) 
084                        return Collections.emptyList();
085                else {
086                        List<BinaryRegion> rns = new ArrayList<BinaryRegion>(regions);
087                        if (sort) {
088                                Collections.sort(rns);
089                        }
090                        return rns;
091                }
092        }
093        
094        // This method must be implemented by any real sub-class:
095        abstract void applyLabeling();
096        
097        // creates a container of BinaryRegion objects
098        // collects the region pixels from the label image
099        // and computes the statistics for each region
100        void collectRegions() {
101                BinaryRegion[] regionArray = new BinaryRegion[maxLabel + 1];
102                for (int i = START_LABEL; i <= maxLabel; i++) {
103                        regionArray[i] = new BinaryRegion(i);
104                }
105                for (int v = 0; v < height; v++) {
106                        for (int u = 0; u < width; u++) {
107                                int label = getLabel(u, v);
108                                if (label >= START_LABEL && label <= maxLabel && regionArray[label] != null) {
109                                        regionArray[label].addPixel(u, v);
110                                }
111                        }
112                }
113                // create a list of regions to return, collect nonempty regions
114                List<BinaryRegion> regionList = new LinkedList<BinaryRegion>();
115                for (BinaryRegion r: regionArray) {
116                        if (r != null && r.getSize() > 0) {
117                                r.update();     // compute the statistics for this region
118                                regionList.add(r);
119                        }
120                }
121                regions = regionList;
122        }
123        
124        public int getLabel(int u, int v) {
125                if (u >= 0 && u < width && v >= 0 && v < height)
126                        return labelArray[u][v];
127                else
128                        return BACKGROUND;
129        }
130        
131        void setLabel(int u, int v, int label) {
132                if (u >= 0 && u < width && v >= 0 && v < height)
133                        labelArray[u][v] = label;
134        }
135        
136        void resetLabel() {
137                currentLabel = -1;
138                maxLabel = -1;
139        }
140        
141        int getNextLabel() {
142                if (currentLabel < 1)
143                        currentLabel = START_LABEL;
144                else
145                        currentLabel = currentLabel + 1;
146                maxLabel = currentLabel;
147                return currentLabel;
148        }
149        
150        int getMaxLabel() {
151                return maxLabel;
152        }

156        public ImageProcessor makeLabelImage(boolean color) {
157                return (color) ?  makeLabelImageColor() : makeLabelImageGray();
158        }
159
160        ColorProcessor makeLabelImageColor() {
161                int[] colorLUT = new int[maxLabel+1];
162                for (int i = START_LABEL; i <= maxLabel; i++) {
163                        colorLUT[i] = makeRandomColor();
164                }
165                ColorProcessor cp = new ColorProcessor(width, height);
166                for (int v = 0; v < height; v++) {
167                        for (int u = 0; u < width; u++) {
168                                int lb = getLabel(u, v);
169                                if (lb >= 0 && lb < colorLUT.length) {
170                                        cp.putPixel(u, v, colorLUT[lb]);
171                                }
172                        }
173                }
174                return cp;
175        }
176        
177        ShortProcessor makeLabelImageGray() {
178                ShortProcessor sp = new ShortProcessor(width, height);
179                for (int v = 0; v < height; v++) {
180                        for (int u = 0; u < width; u++) {
181                                int lb = getLabel(u, v);
182                                sp.set(u, v, (lb >= 0) ? lb : 0);
183                        }
184                }
185                sp.resetMinAndMax();
186                return sp;
187        }
188        
189        // Find the region object with the given label:
190        public BinaryRegion findRegion(int label) {
191                if (label <= 0 || regions == null) return null;
192                BinaryRegion regn = null;
193                for (BinaryRegion r : regions) {
194                        if (r.getLabel() == label) {
195                                regn = r;
196                                break;
197                        }
198                }
199                return regn;
200        }
201        
202        /**
203         * Finds the {@link BinaryRegion} instance associated with
204         * the given image position.
205         * @param u horizontal position
206         * @param v vertical position
207         * @return The associated {@link BinaryRegion} object or null if
208         *              this {@link RegionLabeling} has no region at the given position.
209         */
210        public BinaryRegion getRegion(int u, int v) {
211                int label = getLabel(u, v);
212                return findRegion(label);
213        }

220        int makeRandomColor() {
221                double saturation = 0.2;
222                double brightness = 0.2;
223                float h = (float) Math.random();
224                float s = (float) (saturation * Math.random() + 1 - saturation);
225                float b = (float) (brightness * Math.random() + 1 - brightness);
226                return Color.HSBtoRGB(h, s, b);
227        }

229        void printSummary() {
230                if (regions != null) {
231                        IJ.log("Number of regions detected: " + regions.size());
232                        for (BinaryRegion r : regions) {
233                                IJ.log(r.toString());
234                        }
235                } else
236                        IJ.log("No regions found.");
237        }

253        protected class RegionPixelIterator implements Iterator<Point> {
254                final int label;                                        // the corresponding region's label
255                final int uMin, uMax, vMin, vMax;       // coordinates of region's bounding box
256                int uCur, vCur;                                         // current pixel position
257                Point pNext;                                            // coordinates of the next region pixel
258                boolean first;                                          // control flag
259
260                RegionPixelIterator(BinaryRegion R) {
261                        label = R.getLabel();
262                        first = true;
263                        Rectangle bb = R.getBoundingBox();
264                        uMin = bb.x;
265                        uMax = bb.x + bb.width;
266                        vMin = bb.y;
267                        vMax = bb.y + bb.height;
268                        uCur = uMin;
269                        vCur = vMin;
270                        pNext = null;
271                }
272
273                /** 
274                 * Search from position (uCur, vCur) for the next valid region pixel.
275                 * Return the next position as a Point or null if no such point can be found.
276                 * Don't assume that (uCur, vCur) is a valid region pixel!
277                 * 
278                 * @return the next point
279                 */
280                private Point findNext() {
281                        // start search for next region pixel at (u,v):
282                        int u = (first) ? uCur : uCur + 1;
283                        int v = vCur;
284                        first = false;
285                        while (v <= vMax) {
286                                while (u <= uMax) {
287                                        if (getLabel(u, v) == label) { // next pixel found (uses surrounding labeling)
288                                                uCur = u;
289                                                vCur = v;
290                                                return new Point(uCur, vCur);
291                                        }
292                                        u++;
293                                }
294                                v++;
295                                u = uMin;
296                        }
297                        uCur = uMax + 1;        // just to make sure we'll never enter the loop again
298                        vCur = vMax + 1;
299                        return null;            // no next pixel found
300                }
301
302                /**
303                 * Returns true if the iteration has more elements. (In other words, returns true if next() 
304                 * would return an element rather than throwing an exception.)
305                 * If hasNext() returns true, then pNext refers to a valid region pixel.
306                 */
307                public boolean hasNext() {
308                        if (pNext != null) {    // next element has been queried before but not consumed
309                                return true;
310                        }
311                        else {
312                                pNext = findNext();     // keep next pixel coordinates in pNext
313                                return (pNext != null);
314                        }
315                }
316
317                /**
318                 * Returns: the next element in the iteration
319                 * Throws: NoSuchElementException - if the iteration has no more elements.
320                 */
321                public Point next() {
322                        if (pNext != null || hasNext()) {
323                                Point pn = pNext;
324                                pNext = null;           // "consume" pNext
325                                return pn;
326                        }
327                        else {
328                                throw new NoSuchElementException();
329                        }
330                }
331
332                public void remove() {
333                        throw new UnsupportedOperationException();
334                }
335
336        } // end of RegionPixelIterator

339        /**
340         * This inner class of {@link RegionLabeling} represents a connected 
341         * component or binary region. 
342         * It supports iteration over the contained points, e.g., by
343         * <pre>
344         * BinaryRegion R = ...;
345         * for (Point p : R) {
346         *    // process p ...
347         * }
348         * </pre>
349         */
350        public class BinaryRegion implements Comparable<BinaryRegion>, Iterable<Point> {
351                private final int label;                                // the label of THIS region
352                private int size = 0;
353                private double xc = Double.NaN;
354                private double yc = Double.NaN;
355                private int left = Integer.MAX_VALUE;
356                private int right = -1;
357                private int top = Integer.MAX_VALUE;
358                private int bottom = -1;
359                
360                private Contour outerContour;
361                private List<Contour> innerContours;
362                
363                // summation variables used for various statistics
364                private long x1Sum  = 0;
365                private long y1Sum  = 0;
366                private long x2Sum = 0;
367                private long y2Sum = 0;
368                
369                public double getXc() {
370                        return xc;
371                }
372
373                public double getYc() {
374                        return yc;
375                }
376                
377                public long getX1Sum() {
378                        return x1Sum;
379                }
380
381                public long getY1Sum() {
382                        return y1Sum;
383                }
384
385                public long getX2Sum() {
386                        return x2Sum;
387                }
388
389                public long getY2Sum() {
390                        return y2Sum;
391                }
392                
393                // ------- constructor --------------------------
394                
395                private BinaryRegion(int label){
396                        this.label = label;
397                        this.outerContour = null;
398                        this.innerContours = null;
399                }
400
401                // ------- public methods --------------------------
402                
403                public int getLabel() {
404                        return this.label;
405                }
406                
407                public int getSize() {
408                        return this.size;
409                }
410                
411                public Rectangle getBoundingBox() {
412                        if (right < 0)
413                                return null;
414                        else
415                                return new Rectangle(left, top, right-left + 1, bottom - top + 1);
416                }
417                
418                /**
419                 * Returns the centroid of this region as a 2D point.
420                 * Use {@link getCenterPoint} or {@link getXc} and {@link getYc} instead.
421                 * @deprecated
422                 * @return the centroid of this region
423                 */
424                public Point2D getCenter() {
425                        //TODO: rename to getCenterPoint, replace by returning array, make getters for xc, yc
426                        if (Double.isNaN(xc))
427                                return null;
428                        else
429                                return new Point2D.Double(xc, yc);
430                }
431                
432                
433                /**
434                 * Returns the centroid of this region as a 2D point.
435                 * See also {@link getXc}, {@link getYc}.
436                 * 
437                 * @return the centroid of this region
438                 */
439                public Point2D getCenterPoint() {
440                        //TODO: rename to getCenterPoint, replace by returning array, make getters for xc, yc
441                        if (Double.isNaN(xc))
442                                return null;
443                        else
444                                return new Point2D.Double(xc, yc);
445                }

449                @Override
450                public Iterator<Point> iterator() {
451                        return new RegionPixelIterator(this);
452                }
453                
454                /**
455                 * @deprecated
456                 * Replaced by BinaryRegion implementing {@code Iterable<Point>}.
457                 * Use this method to iterate over all pixels of a region R
458                 * as follows:
459                 * <pre>for (Point p : R.getRegionPoints()) { ... }</pre>
460                 * @return iterator to be used in a for-loop. 
461                 */
462                public Iterable<Point> getRegionPoints() {
463                        //return RegionLabeling.this.getRegionPoints(this);
464                        return new Iterable<Point>() {  // anonymous class!
465                                public Iterator<Point> iterator() {
466                                        return new RegionPixelIterator(BinaryRegion.this);
467                                }
468                        };
469                }
470                
471
472                /**
473                 * Use this method to add a single pixel to this region. Updates summation
474                 * and boundary variables used to calculate various region statistics.
475                 * 
476                 * @param u x-position
477                 * @param v y-position
478                 */
479                protected void addPixel(int u, int v) {
480                        size = size + 1;
481                        x1Sum = x1Sum + u;
482                        y1Sum = y1Sum + v;
483                        x2Sum = x2Sum + u * u;
484                        y2Sum = y2Sum + v * v;
485                        if (u < left)   left = u;
486                        if (v < top)    top = v;
487                        if (u > right)  right = u;
488                        if (v > bottom) bottom = v;
489                }
490                
491                /**
492                 * Call this method to update the region's statistics. For now only the
493                 * center coordinates (xc, yc) are updated. Add additional statements as
494                 * needed to update your own region statistics.
495                 */
496                protected void update() {
497                        if (size > 0) {
498                                xc = (double) x1Sum / size;
499                                yc = (double) y1Sum / size;
500                        }
501                }
502                
503                public Contour getOuterContour() {
504                        return outerContour;
505                }
506                
507                protected void setOuterContour(Contour contr) {
508                        outerContour = contr;
509                }
510                
511                public List<Contour> getInnerContours() {
512                        return innerContours;
513                }
514                
515                protected void makeInnerContours() {
516                        if (innerContours == null) {
517                                innerContours = new LinkedList<Contour>();
518                        }
519                }
520                
521                protected void addInnerContour(Contour contr) {
522                        makeInnerContours();
523                        innerContours.add(contr);
524                }
525                
526                public String toString() {
527                        Formatter fm = new Formatter(new StringBuilder(), Locale.US);
528                        fm.format("Region %d", label);
529                        fm.format(", area = %d", size);
530                        fm.format(", bounding box = (%d, %d, %d, %d)", left, top, right, bottom );
531                        fm.format(", centroid = (%.2f, %.2f)", xc, yc);
532                        if (innerContours != null)
533                                fm.format(", holes = %d", innerContours.size());
534                        String s = fm.toString();
535                        fm.close();
536                        return s;
537                }
538                
539                /**
540                 * Compare method for sorting by region size (larger regions at front)
541                 */
542                public int compareTo(BinaryRegion r2) {
543                        return r2.size - this.size;
544                }

546                /* EXPERIMENTAL STUFF for attaching region properties dynamically:
547                 * Properties can be used to hash results of region calculations
548                 * to avoid multiple calculations.
549                 * Currently, only 'double' values are supported.
550                 * 
551                 * E.g. calculate major axis angle theta for region r, then do
552                 *    r.setProperty("angle", theta);
553                 * and subsequently
554                 *    double theta = r.getProperty("angle");
555                 */
556                
557                private Map<String, Double> properties = null;
558                
559                /**
560                 * Sets the specified property of this region to the given value.
561                 * @param name Chosen name of the property.
562                 * @param val Value associated with this property.
563                 */
564                public void setProperty(String name, double val) {
565                        if (properties == null) {
566                                properties = new HashMap<String, Double>();
567                        }
568                        properties.put(name, val);
569                }
570                
571                /**
572                 * Retrieves the specified region property. 
573                 * An {@link IllegalArgumentException} is thrown if the property 
574                 * is not defined for this region.
575                 * @param name The name of the property.
576                 * @return The value associated with the specified property.
577                 */
578                public double getProperty(String name) {
579                        Double value;
580                        if (properties == null || (value = properties.get(name)) == null) {
581                                throw new IllegalArgumentException("Region property " + name + " is undefined.");
582                        }
583                        return value.doubleValue();
584                }
585                
586                /**
587                 * Removes all properties attached to this region.
588                 */
589                public void clearProperties() {
590                        properties.clear();
591                }
