010package imagingbook.pub.geometry.mappings.linear;
011
012
013public class Rotation extends AffineMapping {
014        
015        /**
016         * Creates a new {@link AffineMapping} representing a pure 2D rotation
017         * around the origin.
018         * @param alpha Rotation angle (in radians).
019         */
020        public Rotation(double alpha) {
021                super(
022                         Math.cos(alpha), -Math.sin(alpha), 0,
023                         Math.sin(alpha),  Math.cos(alpha), 0, false);
024        }
025}
