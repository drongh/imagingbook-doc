001package imagingbook.pub.geometry.points;
002
003import java.util.List;
004
005import org.apache.commons.math3.linear.MatrixUtils;
006import org.apache.commons.math3.linear.RealMatrix;
007
008import imagingbook.lib.math.Matrix;
009
010
011/**
012 * This class implements the Iterative Closest Point algorithm for
013 * n-dimensional samples.
014 * 
015 * @author W. Burger
016 *
017 */
018public class IterativeClosestPointMatcher {
019        
020        private final double tau;
021        private final int kMax;
022        private final int mx, my, n;
023        private final int[] A;
024        
025        private int k = 0;
026        private double eMin = Double.POSITIVE_INFINITY;
027        private RealMatrix T;
028        private boolean converged = false;
029        
030        
031        public IterativeClosestPointMatcher(List<double[]> X, List<double[]> Y, double tau, int kMax) {
032                this.tau = tau;
033                this.kMax = kMax;
034                this.mx = X.size();
035                this.my = Y.size();
036                this.n = X.get(0).length;
037                this.A = new int[mx];
038                
039                System.out.println("X length = " + mx);
040                System.out.println("Y length = " + my);
041                System.out.println("X dimension = " + X.get(0).length);
042                
043                match(X, Y);
044        }
045        
046        private void match(List<double[]> X, List<double[]> Y) {
047                k = 0;
048                eMin = Double.POSITIVE_INFINITY;
049                T = initialTransformation();
050                
051                do {
052                        System.out.println("k = " + k);
053                        System.out.println("T = " + Matrix.toString(T.getData()));
054                        associatePoints(X, Y);
055                        double e = fitPoints(X, Y);     // sets T
056                        double de = eMin - e;
057                        converged = (0 <= de && de < tau);
058                        eMin = e;
059                        k = k + 1;
060                } while(!converged && k <= kMax);
061        }
062
063        private RealMatrix initialTransformation() {
064                RealMatrix iT = MatrixUtils.createRealMatrix(n, n + 1);
065                iT.setSubMatrix(MatrixUtils.createRealIdentityMatrix(n).getData(), 0, 0);
066                return iT;
067        }
068        
069        private void associatePoints(List<double[]> X, List<double[]> Y) {
070                int i = 0;
071                for (double[] xi : X) {
072                        double[] xiT = T.operate(Matrix.toHomogeneous(xi)); // T.applyTo(xi);
073                        double dMin = Double.POSITIVE_INFINITY;
074                        int jMin = 0;
075                        int j = 0;
076                        for (double[] yi : Y) {
077                                double d = Matrix.normL2squared(Matrix.subtract(xiT,yi));
078                                if (d < dMin) {
079                                        dMin = d;
080                                        jMin = j;
081                                }
082                                j = j + 1;
083                        }
084                        A[i] = jMin;
085                        i = i + 1;
086                }
087                
088        }
089
090        private double fitPoints(List<double[]> X, List<double[]> Y) {
091                ProcrustesFit pf = new ProcrustesFit(X, Y);
092                T = pf.getTransformationMatrix();
093                return pf.getError();
094        }
095
096        public boolean hasConverged() {
097                return converged;
098        }
099
100}
