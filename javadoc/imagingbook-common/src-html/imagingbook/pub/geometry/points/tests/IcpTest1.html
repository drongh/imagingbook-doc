001package imagingbook.pub.geometry.points.tests;
002
003import java.util.ArrayList;
004import java.util.List;
005import java.util.Random;
006
007import imagingbook.pub.geometry.mappings.linear.AffineMapping;
008import imagingbook.pub.geometry.mappings.linear.Rotation;
009import imagingbook.pub.geometry.mappings.linear.Translation;
010import imagingbook.pub.geometry.points.IterativeClosestPointMatcher;
011
012public class IcpTest1 {
013        
014        static int m = 50;
015        static int size = 100;
016        static double theta = 0.1;
017        static double dx = 4;
018        static double dy = -6;
019        
020        static double sigma = 2.5;
021
022        
023        static double tau = 0.1;
024        static int kMax = 20;
025        
026        AffineMapping A = null;
027        List<double[]> X, Y;
028        
029        Random rnd = new Random(11);
030
031        public static void main(String[] args) {
032                // TODO Auto-generated method stub
033                new IcpTest1().run();
034        }
035        
036        void run() {
037                A = makeTransformation();
038                makeSamplePointsX();
039                makeSamplePointsY();
040                IterativeClosestPointMatcher icp = 
041                                new IterativeClosestPointMatcher(X, Y, tau, kMax);
042                
043                System.out.println("ICP has converged: " + icp.hasConverged());
044        }
045
046        private AffineMapping makeTransformation() {
047                double ctr = 0.5 * size;
048                AffineMapping am = new AffineMapping();
049                am.concatDestructive(new Translation(-ctr, -ctr));      // TODO: rename to concatD
050                am.concatDestructive(new Rotation(theta));
051                am.concatDestructive(new Translation(ctr, ctr));
052                am.concatDestructive(new Translation(dx, dy));
053                return am;
054        }
055
056        private void  makeSamplePointsX() {
057                X = new ArrayList<double[]>(m);
058                for (int i = 0; i < m; i++) {
059                        double x = rnd.nextInt(size);
060                        double y = rnd.nextInt(size);
061                        X.add(new double[] {x, y});
062                }
063        }
064
065        private void makeSamplePointsY() {
066                Y = new ArrayList<double[]>(m);
067                for (double[] xi : X) {
068                        double[] xiT = A.applyTo(xi);
069                        xiT[0] = xiT[0] + sigma * rnd.nextGaussian();
070                        xiT[1] = xiT[1] + sigma * rnd.nextGaussian();
071                        Y.add(xiT);
072                }
073                
074        }
075}
