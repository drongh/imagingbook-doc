001package imagingbook.lib.math;
002
003import java.util.List;
004
005import org.apache.commons.math3.linear.ArrayRealVector;
006import org.apache.commons.math3.linear.LUDecomposition;
007import org.apache.commons.math3.linear.MatrixUtils;
008import org.apache.commons.math3.linear.RealMatrix;
009import org.apache.commons.math3.linear.RealVector;
010import org.apache.commons.math3.linear.SingularValueDecomposition;
011
012import imagingbook.pub.geometry.mappings.linear.AffineMapping;
013
014/**
015 * This class implements the n-dimensional Procrustes fit algorithm described in 
016 * Shinji Umeyama, "Least-squares estimation of transformation parameters 
017 * between two point patterns", IEEE Transactions on Pattern Analysis and 
018 * Machine Intelligence 13.4 (Apr. 1991), pp. 376â€“380.
019 * 
020 * @author W. Burger
021 * @version 2017/03/01
022 *
023 */
024public class ProcrustesFit {
025        
026        final boolean allowTranslation;
027        final boolean allowScaling;
028        final boolean forceRotation;
029        
030        final int m;            // number of samples
031        final int n;            // dimension of samples
032        
033        final double c;         // scale
034        final RealMatrix Q;     // orthogonal (rotation) matrix
035        final RealVector t;     // translation vector
036        final double err;       // total (squared) error
037        
038        // --------------------------------------------------------------
039        
040        /**
041         * Convenience constructor. The supplied point sequences must have the
042         * same length and order (i.e., points must be in correspondence).
043         * @param xA Sequence of n-dimensional points
044         * @param xB Sequence of n-dimensional points (reference)
045         */
046        public ProcrustesFit(List<double[]> xA, List<double[]> xB) {
047                this(xA, xB, true, true, true);
048        }
049        
050        /**
051         * Full constructor. The supplied point sequences must have the
052         * same length and order (i.e., points must be in correspondence).
053         * @param xA Sequence of n-dimensional points
054         * @param xB Sequence of n-dimensional points (reference)
055         * @param allowTranslation If {@code true}, translation (t) between point sets is considered, 
056         *              otherwise zero translation is assumed.
057         * @param allowScaling If {@code true}, scaling (c) between point sets is considered, 
058         *              otherwise unit scale assumed.
059         * @param forceRotation If {@code true}, the orthogonal part of the transformation (Q)
060         *              is forced to a true rotation and no reflection is allowed.
061         */
062        public ProcrustesFit(List<double[]> xA, List<double[]> xB, 
063                        boolean allowTranslation, boolean allowScaling, boolean forceRotation) {
064                if (xA.size() != xB.size())
065                        throw new IllegalArgumentException("point sequences xA, xB must have same length");
066                
067                this.allowTranslation = allowTranslation;
068                this.allowScaling = allowScaling;
069                this.forceRotation = forceRotation;
070                this.m = xA.size();
071                this.n = xA.get(0).length;
072                
073                double[] meanA = null;
074                double[] meanB = null;
075                
076                if (this.allowTranslation) {
077                        meanA = getMeanVec(xA);
078                        meanB = getMeanVec(xB);
079                }
080                
081                RealMatrix A = makeDataMatrix(xA, meanA);
082                RealMatrix B = makeDataMatrix(xB, meanB);
083                MatrixUtils.checkAdditionCompatible(A, B);      // A, B of same dimensions?
084                
085                RealMatrix BAt = B.multiply(A.transpose());
086                SingularValueDecomposition svd = new SingularValueDecomposition(BAt);
087                
088                RealMatrix U = svd.getU();
089                RealMatrix S = svd.getS();
090                RealMatrix V = svd.getV();
091                        
092                double d = (svd.getRank() >= n) ? det(BAt) : det(U) * det(V);
093                
094                RealMatrix D = MatrixUtils.createRealIdentityMatrix(n);
095                if (d < 0 && forceRotation)
096                        D.setEntry(n - 1, n - 1, -1);
097                
098                Q = U.multiply(D).multiply(V.transpose());
099                
100                double normA = A.getFrobeniusNorm();
101                double normB = B.getFrobeniusNorm();
102                
103                c = (this.allowScaling) ? 
104                                S.multiply(D).getTrace() / sqr(normA) : 1.0;
105                
106                if (allowTranslation) {
107                        RealVector ma = MatrixUtils.createRealVector(meanA);
108                        RealVector mb = MatrixUtils.createRealVector(meanB);
109                        t = mb.subtract(Q.scalarMultiply(c).operate(ma));
110                }
111                else {
112                        t = new ArrayRealVector(n);     // zero vector
113                }
114                
115                err = sqr(normB) - sqr(S.multiply(D).getTrace() / normA);
116        }
117        
118        // -----------------------------------------------------------------
119        
120        /**
121         * Retrieves the estimated scale.
122         * @return The estimated scale (or 1 if {@code allowscaling = false}).
123         */
124        public double getScale() {
125                return c;
126        }
127        
128        /**
129         * Retrieves the estimated orthogonal (rotation) matrix.
130         * @return The estimated rotation matrix.
131         */
132        public RealMatrix getQ() {
133                return Q;
134        }
135        
136        /**
137         * Retrieves the estimated translation vector.
138         * @return The estimated translation vector.
139         */
140        public RealVector getT() {
141                return t;
142        }
143        
144        /**
145         * Retrieves the total (squared) error for the estimated fit.
146         * @return The total error for the estimated fit.
147         */
148        public double getError() {
149                return err;
150        }
151        
152        // calculate total error (for testing)
153        /**
154         * Calculates the total error for the estimated fit as
155         * the sum of the squared Euclidean distances between the 
156         * transformed point set A and the reference set B.
157         * This method is provided for testing as an alternative to
158         * the quicker {@link getError} method.
159         * @param xA Sequence A of n-dimensional points.
160         * @param xB Sequence B of n-dimensional points (reference).
161         * @return The total error for the estimated fit.
162         */
163        public double getEuclideanError(List<double[]> xA, List<double[]> xB) {
164                RealMatrix cQ = Q.scalarMultiply(c);
165                double ee = 0;
166                for (int i = 0; i < xA.size(); i++) {
167                        RealVector ai = new ArrayRealVector(xA.get(i));
168                        RealVector bi = new ArrayRealVector(xB.get(i));
169                        RealVector aiT = cQ.operate(ai).add(t);
170                        double ei = aiT.subtract(bi).getNorm();
171                        ee = ee + sqr(ei);
172                }
173                return ee;
174        }
175        
176        public AffineMapping getAffineMapping2D() {
177                if (n != 2)
178                        return null;
179                AffineMapping map = new AffineMapping(
180                                c * Q.getEntry(0, 0), c * Q.getEntry(0, 1), t.getEntry(0),
181                                c * Q.getEntry(1, 0), c * Q.getEntry(1, 1), t.getEntry(1),
182                                false
183                                );
184                return map;     
185        }
186        
187        // -----------------------------------------------------------------
188        
189        private double det(RealMatrix M) {
190                return new LUDecomposition(M).getDeterminant();
191        }
192        
193        private double sqr(final double x) {
194                return x * x;
195        }
196        
197        private double[] getMeanVec(List<double[]> X) {
198                double[] sum = new double[X.get(0).length];
199                for (double[] x : X) {
200                        for (int j = 0; j < x.length; j++) {
201                                sum[j] = sum[j] + x[j];
202                        }
203                }
204                Matrix.multiplyD(1.0 / X.size(), sum);
205                return sum;
206        }
207        
208        private RealMatrix makeDataMatrix(List<double[]> X) {
209                final int m = X.size();
210                final int n = X.get(0).length;
211                RealMatrix M = MatrixUtils.createRealMatrix(n, m);
212                int i = 0;
213                for (double[] x : X) {
214                        RealVector xi = MatrixUtils.createRealVector(x);
215                        M.setColumnVector(i, xi);
216                        i++;
217                }
218                return M;
219        }
220        
221        private RealMatrix makeDataMatrix(List<double[]> X, double[] meanX) {
222                if (meanX == null) {
223                        return makeDataMatrix(X);
224                }
225                final int m = X.size();
226                final int n = X.get(0).length;
227                RealMatrix M = MatrixUtils.createRealMatrix(n, m);
228                RealVector mean = MatrixUtils.createRealVector(meanX);
229                int i = 0;
230                for (double[] x : X) {
231                        RealVector xi = MatrixUtils.createRealVector(x).subtract(mean);
232                        M.setColumnVector(i, xi);
233                        i++;
234                }
235                return M;
236        }
237        
238        private void printSVD(SingularValueDecomposition svd) {
239                RealMatrix U = svd.getU();
240                RealMatrix S = svd.getS();
241                RealMatrix V = svd.getV();
242                System.out.println("------ SVD ---------------");
243                System.out.println("U = " + Matrix.toString(U.getData()));
244                System.out.println("S = " + Matrix.toString(S.getData()));
245                System.out.println("V = " + Matrix.toString(V.getData()));
246                System.out.println("--------------------------");
247        }
