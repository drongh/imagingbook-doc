007public abstract class ProcrustesExample {
008        
009        public abstract void run();
010        
011        protected double roundToDigits(double x, int ndigits) {
012                int d = (int) Math.pow(10, ndigits);
013                return Math.rint(x * d) / d;
014        }
015        
016        
017        protected void showHeadline(String title) {
018                System.out.println("\n**************** " + title + " ****************");
019        }
020        
021        protected void print(List<double[]> lX) {
022                for (double[] x : lX) {
023                        System.out.println("   " + Matrix.toString(x));
024                }
025        }
026        
027        
028        public static void main(String[] args) {
029                PrintPrecision.set(10);
030                new ProcrustesExample1().run();
031                new ProcrustesExample2().run();
032        }
033
034
035}
